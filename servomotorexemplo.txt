#include <Servo.h>

Servo servo1;

void setup() {
    servo1.attach(4);  // Attach the servo to pin 4
}

void loop() {
    int angulo = analogRead(A0);               // Read from analog pin A0
    angulo = map(angulo, 0, 1023, 0, 180);    // Map 0-1023 to 0-180 degrees
    servo1.write(angulo);                      // Set servo position
    delay(15);                                 // Small delay for smooth movement
}
